    alpha = min(0.8, alpha + 0.2)  # 提高乘客优先级

## 使用指南

### 快速开始
```bash
# 1. 安装依赖
pip install -r requirements.txt

# 2. 运行主程序
python passenger_cargo_optimization.py

# 3. 运行示例
python example_usage.py
```

### 自定义使用
```python
from passenger_cargo_optimization import *

# 定义节点
nodes = [
    Node(id=0, name="depot", node_type=NodeType.DEPOT, location=(0,0)),
    Node(id=1, name="村庄A", node_type=NodeType.PASSENGER, 
         location=(5,3), demand_passenger=3),
    # ... 更多节点
]

# 定义车辆
vehicles = [
    Vehicle(id=0, name="1号车", capacity_passenger=20, capacity_cargo=200)
]

# 创建优化器
optimizer = PassengerCargoOptimizer(nodes, vehicles)

# 执行优化
routes = optimizer.optimize_with_ai_coordination(max_iterations=100)
```
